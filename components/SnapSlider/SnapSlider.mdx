export const SnapSlider = ({ children, className, ...props }) => (
  <div
    {...props}
    className={`flex items-center x-gap-3 rounded-lg snap-x snap-mandatory overflow-x-scroll max-w-200 mx-auto border border-neutral-400 ${className}`}
    aria-roledescription="carousel"
  >
    {children?.map((el, i) => (
      <div
        className="snap-center flex-shrink-0 min-w-120 max-w-200"
        role="tabpanel"
        aria-roledescription="slide"
        aria-label={`${i + 1} of ${children.length}`}
      >
        {el}
      </div>
    ))}
  </div>
);
