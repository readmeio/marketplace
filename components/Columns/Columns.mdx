import React, { Children } from 'react';

export const Column = ({ children }) => {
  return <div className="Column">{children}</div>;
};

export const Columns = ({ children, layout = 'auto' }) => {
  const columnsCount = Children.count(children);
  const layoutType = layout === 'fixed' ? '1fr' : 'auto';

  return (
     <div className={`grid gap-5 grid-cols-[repeat(${columnsCount},${layoutType})] max-[414px]:grid-cols-1`}>
      {children}
    </div>
  );
};
